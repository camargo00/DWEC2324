function generarCaminata(minutosDisponibles) { // Direcciones posibles: 'n' (norte), 's' (sur), 'o' (oeste), 'e' (este) const direcciones = ['n', 's', 'o', 'e']; // Función auxiliar para obtener una dirección aleatoria function obtenerDireccionAleatoria() { const indiceAleatorio = Math.floor(Math.random() * direcciones.length); return direcciones[indiceAleatorio]; } // Generar la secuencia de direcciones let secuencia = ''; for (let i = 0; i < minutosDisponibles; i++) { const direccion = obtenerDireccionAleatoria(); secuencia += direccion; } return secuencia; } // Ejemplo de uso con 10 minutos disponibles const minutosParaPaseo = 10; const secuenciaPaseo = generarCaminata(minutosParaPaseo); console.log(`Secuencia de paseo para ${minutosParaPaseo} minutos: ${secuenciaPaseo}`);
